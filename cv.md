## Code example
#### Unique In Order Task from Codewars, 6 kyu
```
const uniqueInOrder = function(iterable){
  let arr = [...iterable]
  let result = arr.slice(0,1)

  for (let i of arr) {
    let last = result[(result.length-1)]
    if (i !== last) {
      result.push(i)
    }
  }
  return result
}
```
